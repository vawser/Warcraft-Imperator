add_correct_trait_effect = {
	if = {
		limit = {
			has_variable = which_disease
			var:which_disease = 1
			NOT = {
				has_trait = mentagra
			}
		}
		add_trait = mentagra
	}
	else_if = {
		limit = {
			has_variable = which_disease
			var:which_disease = 2
			NOT = {
				has_trait = arthritis
			}
		}
		add_trait = arthritis
	}
	else_if = {
		limit = {
			has_variable = which_disease
			var:which_disease = 3
			NOT = {
				has_trait = pox
			}
		}
		add_trait = pox
	}
	else_if = {
		limit = {
			has_variable = which_disease
			var:which_disease = 4
			NOT = {
				has_trait = dysentery
			}
		}
		add_trait = dysentery
	}
	else_if = {
		limit = {
			has_variable = which_disease
			var:which_disease = 5
			NOT = {
				has_trait = infection
			}
		}
		add_trait = infection
	}
	else_if = {
		limit = {
			has_variable = which_disease
			var:which_disease = 6
			NOT = {
				has_trait = inflammation
			}
		}
		add_trait = inflammation
	}
	if = {
		limit = {
			has_variable = which_disease
		}
		remove_variable = which_disease
	}
}

add_result_affliction_effect = {
	if = {
		limit = {
			has_variable = medical_result_acquired
			var:medical_result_acquired = 1
			NOT = {
				has_trait = barren
			}
		}
		add_trait = barren
	}
	else_if = {
		limit = {
			has_variable = medical_result_acquired
			var:medical_result_acquired = 2
			NOT = {
				has_trait = castrated
			}
		}
		add_trait = castrated
	}
	else_if = {
		limit = {
			has_variable = medical_result_acquired
			var:medical_result_acquired = 3
			NOT = {
				has_trait = frail
			}
		}
		add_trait = frail
	}
	else_if = {
		limit = {
			has_variable = medical_result_acquired
			var:medical_result_acquired = 4
			NOT = {
				has_trait = blind
			}
		}
		if = {
			limit = { has_trait = one_eyed }
			remove_trait = one_eyed
			add_trait = blind
		}
		else = {
			add_trait = one_eyed
		}
	}
	else_if = {
		limit = {
			has_variable = medical_result_acquired
			var:medical_result_acquired = 5
			NOT = {
				has_trait = paralysis
			}
		}
		add_trait = paralysis
	}
	else_if = {
		limit = {
			has_variable = medical_result_acquired
			var:medical_result_acquired = 6
			NOT = {
				has_trait = blind
			}
		}
		if = {
			limit = { has_trait = one_eyed }
			remove_trait = one_eyed
		}	
		add_trait = blind
	}
	else_if = {
		limit = {
			has_variable = medical_result_acquired
			var:medical_result_acquired = 7
			NOT = {
				has_trait = brain_damaged
			}
		}
		add_trait = brain_damaged
	}
	if = {
		limit = {
			has_variable = medical_result_acquired
		}
		remove_variable = medical_result_acquired
	}
}

randomise_result_from_treatment_trait_effect = {
	if = {
		limit = { var:medical_accident_problem = 1 }
		add_trait = barren
	}
	else_if = {
		limit = { var:medical_accident_problem = 2 }
		add_trait = castrated
	}
	else_if = {
		limit = { var:medical_accident_problem = 3 }
		if = {
			limit = { has_trait = one_eyed }
			remove_trait = one_eyed
		}
		add_trait = blind
	}
	else_if = {
		limit = { var:medical_accident_problem = 4 }
		add_trait = maimed
	}
	else_if = {
		limit = { var:medical_accident_problem = 5 }

		}

	}
	else_if = {

		add_trait = frail
	}
	else_if = {
		limit = { var:medical_accident_problem = 7 }
		add_trait = brain_damaged
	}
}
