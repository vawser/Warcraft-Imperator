support_successor = {

	on_other_nation = yes
	on_own_nation = no
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_corruption"
	
	potential_trigger = {
		hidden:scope:target = {
			is_alive = yes
			is_mercenary = no
			employer = {
				NOT = { tag = MER }
				NOT = { tag = BAR }
				NOT = { tag = REB }
				NOT = { tag = PIR }
				is_monarchy = yes
				is_subject = no
				NOT = {
					any_character = {
						has_variable = successor_support_ci
					}
				}
			}
			NOT = {
				employer = scope:actor
			}
		}
	}
	
	
	allowed_trigger = {
		hidden:scope:target = {
			is_pretender = yes
			is_adult = yes
			NOT = { has_character_modifier = disloyal_successor_cmod }
		}
		scope:actor = {
			in_diplomatic_range = scope:target.employer
		}
		hidden:scope:actor = {
			treasury >= 1000
			political_influence >= 100
		}
		hidden:scope:target = {
			is_ruler = no
			is_primary_heir = no
			loyalty <= 50
		}
		scope:target = {
			employer = {
				has_civil_war = no
			}
		}
	}
	

	
	effect = {
		scope:actor = {
			add_treasury = -1000
			add_political_influence = -100
		}
		scope:target = {
			add_loyalty = loyalty_pretender_support_ci 
			set_variable = {
				name = successor_support_ci
				value = scope:actor
				days = 1850
			}
		}
		custom_tooltip = "pretender_support_ci_tt"
	}
	
}
