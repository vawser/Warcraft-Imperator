free_hands = {

	on_other_nation = no
	on_own_nation = yes
	message = no
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_give_free_hands"
	
	potential_trigger = {
		hidden:scope:target = {
			is_alive = yes
			is_mercenary = no
			is_ruler = no
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			OR = {
				has_any_office = yes
				has_tech_office = yes
				is_governor = yes
				is_clan_chief = yes
			}
			NOT = {
				has_triggered_character_modifier = free_hands_cmod
			}
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			OR = {
				has_any_office = yes
				has_tech_office = yes
				is_governor = yes
				is_clan_chief = yes
			}
		}
	}
	
	effect = {
		custom_tooltip = free_hands_tooltip
		scope:target = {
			add_triggered_character_modifier = {
				name = free_hands_cmod
				duration = -1
			}
			set_variable = {
				name = free_hands_cooldown
				days = 730
			}
			add_loyalty = given_free_hands
		}
	}
	
	ai_will_do = {
		modifier = {
			add = 10
			trigger = {
				employer = {
					is_tribal = yes
				}
				is_clan_chief = yes
				loyalty <= 50
				corruption <= 40
			}
		}
	}
	
	
}	

tie_hands = {

	on_other_nation = no
	on_own_nation = yes
	message = no
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_give_free_hands"
	
	potential_trigger = {
		hidden:scope:target = {
			has_character_modifier = free_hands_cmod
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			has_triggered_character_modifier = free_hands_cmod
			custom_tooltip = {
				text = free_hands_cooldown_tooltip
				NOT = {
					has_variable = free_hands_cooldown
				}
			}
		}
	}
	
	effect = {
		custom_tooltip = tie_hands_tooltip
		scope:target = {
			remove_triggered_character_modifier = free_hands_cmod
			remove_loyalty = given_free_hands
			add_loyalty = free_hands_revoked
		}
	}
	
	ai_will_do = {
		modifier = {
			add = 10
			trigger = {
				OR = {
					AND = {
						corruption >= 50
						NOT = {
							has_variable = free_hands_cooldown
						}
					}
					AND = {
						loyalty >= 70
						NOT = {
							has_variable = free_hands_cooldown
						}
					}
				}
			}
		}
	}
	
	
	
	
}
