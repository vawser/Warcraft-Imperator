exalt_clan_chief = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_reward"
	
	potential_trigger = {
		hidden:scope:target = {
			is_alive = yes
			is_clan_chief = yes
			is_ruler = no
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
		hidden:scope:actor = {
			any_character = {
				count > 1
				is_clan_chief = yes
			}
		}
	}
	
	
	allowed_trigger = {
		custom_tooltip = {
			text = no_other_clan_chief_modifier_tt
			scope:actor = {
				NOT = {
					any_character = {
						is_clan_chief = yes
						has_triggered_character_modifier = exalted_clan_chief
					}
				}
			}
		}
	}
	
	effect = {
		scope:actor = { 
			add_treasury = {
				value = has_monthly_income
				multiply = -3
			}
		}	
		scope:target = {
			add_loyalty = chief_exalted
			hidden_effect = {
				if = {
					limit = {
						has_triggered_character_modifier = insulted_clan_chief
					}
					remove_triggered_character_modifier = insulted_clan_chief
				}
			}
			add_triggered_character_modifier = {
				name = exalted_clan_chief
				duration = 1825
			}
		}
		custom_tooltip = every_other_clan_chief
		scope:actor = {
			every_character = {
				limit = {
					is_clan_chief = yes
					NOT = {
						this = scope:target
					}
				}
				hidden_effect = {
					if = {
						limit = {
							has_triggered_character_modifier = insulted_clan_chief
						}
						remove_triggered_character_modifier = insulted_clan_chief
					}
					if = {
						limit = {
							has_triggered_character_modifier = exalted_clan_chief
						}
						remove_triggered_character_modifier = exalted_clan_chief
					}
				}
				add_triggered_character_modifier = {
					name = insulted_clan_chief
					duration = 1825
				}
				add_loyalty = insulted_clan_chief
			}
		}
	}
	
}
