prove_legitimacy = {

	on_other_nation = no
	on_own_nation = yes
	
	message = no
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_scholar"
	
	potential_trigger = {
		hidden:scope:target = {
			is_ruler = yes
			is_alive = yes
			employer = {
				is_monarchy = yes
			}
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			trigger_if = { 
				limit = { is_adult = no }
				is_adult = yes
			}
			is_ruler = yes
			wealth >= 300
			NOT = {
				has_ambition = scheme_ruler_legitimacy
			}
		}
	}
	

	
	effect = {
		scope:target = {
			clear_ambition_effect_ci = yes
			set_variable = {
				name = ongoing_scheme_target
				value = scope:target
				days = 770
			} 
			set_ambition = scheme_ruler_legitimacy
		}
		custom_tooltip = legitimacy_ambition_effect_tt
	}
	
	ai_will_do = {
		modifier = {
			add = 5
			trigger = {
				is_ruler = yes
				employer = {
					legitimacy <= 60
				}
			}
		}
	}
	
}
