grant_laurels = {

	on_other_nation = no
	on_own_nation = yes
	message = no
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_punishment"
	
	potential_trigger = {
		hidden:scope:actor = {
			is_republic = yes
		}
		hidden:scope:target = {
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			is_alive = yes
			is_mercenary = no
			is_ruler = no
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			is_general = yes
			loyalty <= 40
			custom_tooltip = {
				text = laurels_already_granted_tt
				NOT = {
					has_variable = already_granted_laurels
				}
			}
		}
		scope:actor = {
			can_pay_price = price_grant_laurels
			stability >= 10
		}
	}
	
	effect = {
		scope:actor = { pay_price = price_grant_laurels }
		scope:target = {
			set_variable = {
				name = already_granted_laurels
			}
			add_loyalty = laurels_acknowledged_l
			add_prominence = 20
			add_popularity = 20
			if = {
				limit = {
					has_nickname = no
				}
				custom_tooltip = may_gain_nickname
				hidden_effect = {
					add_triumph_nickname_effect = yes
				}
			}
		}
		scope:actor = {
			if = {
				limit = {
					stability >= 10
				}
				add_stability = -20
			}
		}
		scope:actor.current_ruler = {
			add_popularity = subtract_popularity_huge
		}
	}
	
	ai_will_do = {
		modifier = {
			add = 5
			trigger = {
				loyalty < 5
				employer = {
					stability >= 40
				}
			}
		}
	}

}
