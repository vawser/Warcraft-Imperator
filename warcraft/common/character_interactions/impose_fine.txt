impose_corruption_sanctions = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_bribe"
	
	potential_trigger = {
		scope:target = {
			is_adult = yes
			is_alive = yes
			is_ruler = no
		}
		hidden:scope:target = {
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			corruption > 5
			NOR = {
				has_trait = crafty
				has_trait = corrupt
				has_character_modifier = populist_demagogue_oh_no
			}
			NOT = { has_loyalty = corruption_sanctions_l }
		}
	}
	

	
	effect = {
		scope:actor = {
			add_tyranny = {
				value = scope:target.corruption
				divide = 10
				min = 1
			}
		}
		scope:target = {
			add_loyalty = corruption_sanctions_l
			add_corruption = {
				value = 0
				subtract = corruption
				max = -5
				min = -50
			}
			if = {
				limit = {
					is_minor_character = no
					family = {
						prestige > 0
					}
				}
				family = {
					add_prestige = {
						value = 0
						subtract = prestige
						divide = 100
						multiply = {
							value = scope:target.corruption
							divide = 4
						}
						max = -10
					}
				}
			}
		}
	}
	
}
