#Hunt pirates unit objective.

objective_hunt_pirates = {
	allow = {
		is_navy = yes
	}
	
	mode = add #none/add/replace
	
	weight = {
		#Don't engage enemies...
		modifier = {
			trigger = {
				scope:province = {
					has_enemy_navy = root
				}
			}
			add = {
				value = 1000000
			}
		}
		#...unless they say YARRRGH.
		modifier = {
			trigger = {
				scope:province = {
					any_unit_in_province = {
						unit_owner = c:PIR
					}
				}
			}
			add = {
				value = -1000000
			}
		}
	}
}
