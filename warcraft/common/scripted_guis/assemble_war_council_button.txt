assemble_war_council_button = {
    scope = country
	saved_scopes = {
		
	}
    is_shown = { # Can be omitted as always true
       OR = {
		is_monarchy = yes
		is_tribal = yes
	   }
    }
	ai_is_valid = {
		war = no
		trigger_if = {
			limit = {
				AND = {
					is_tribal = yes
					is_ai = yes
				}
			}
			is_tribal = no
		}
	}
    is_valid = {
		trigger_if = {
			limit = {
				NOT = {
					any_character = {
						count >= 3
						is_ruler = no
						is_adult = yes
					}
				}
			}
			any_character = {
				count >= 3
				is_ruler = no
				is_adult = yes
			}
		}
		trigger_if = {
			limit = {
				has_variable = summoned_war_council
			}
			custom_tooltip = {
				text = "not_summoned_war_council_recently_tt"
				NOT = {
					has_variable = summoned_war_council
				}
			}
		}
		trigger_if = {
			limit = {
				NOT = {
					any_neighbour_country = {
						NOR = {
							alliance_with = root
							is_subject_of = root
						}
					}
				}
			}
			any_neighbour_country = {
				NOR = {
					alliance_with = root
					is_subject_of = root
				}
			}
		}
    }
    effect = {
    	custom_tooltip = "summon_war_council_desc_tt"
       	set_variable = {
			name = summoned_war_council
			days = 3650
	   	}
	   	trigger_event = {
			id = war_council.1
	   	}
    }
}
